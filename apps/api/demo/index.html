Change the scene ID to any ID returned by the scene index or a collection endpoint.

import { AtmosceneClient } from '/api/v1/client.mjs';

const atmoscene = new AtmosceneClient();
const scene = await atmoscene.scene('day.sunset.none.rain');
await atmoscene.mount('#sky', scene);
